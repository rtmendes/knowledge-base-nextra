#### 3. Videos Table

 CREATE TABLE videos ( id VARCHAR PRIMARY KEY, title VARCHAR NOT NULL, description TEXT, thumbnail_url VARCHAR, channel_id VARCHAR REFERENCES channels(id), view_count INTEGER, like_count INTEGER, comment_count INTEGER, published_at TIMESTAMP WITH TIME ZONE, user_id UUID NOT NULL, created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(), transcription TEXT ); -- Junction table for playlists and videos CREATE TABLE playlist_videos ( playlist_id VARCHAR REFERENCES playlists(id), video_id VARCHAR REFERENCES videos(id), position INTEGER, PRIMARY KEY (playlist_id, video_id) ); -- Enable Row Level Security ALTER TABLE videos ENABLE ROW LEVEL SECURITY; ALTER TABLE playlist_videos ENABLE ROW LEVEL SECURITY; -- Create policies CREATE POLICY "Users can view their own videos" ON videos FOR ALL USING (auth.uid() = user_id); CREATE POLICY "Users can view their own playlist_videos" ON playlist_videos FOR ALL USING ( EXISTS ( SELECT 1 FROM playlists WHERE playlists.id = playlist_videos.playlist_id AND playlists.user_id = auth.uid() ) );
