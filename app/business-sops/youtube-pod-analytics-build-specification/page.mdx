# YouTube POD Analytics

## Vercel AI Build Specification (April 30, 2025)

## Introduction

This document provides comprehensive specifications for Vercel AI to build the YouTube POD Analytics platform, a system designed to identify print-on-demand opportunities from YouTube comments, analyze marketplace trends, and generate product designs.

#### Important Implementation Note

This specification is structured specifically for Vercel AI implementation. All components, API routes, database schemas, and UI elements are defined in a format that can be directly translated into code.

## System Architecture Overview

The YouTube POD Analytics platform follows a modern Next.js architecture with the following key components:

#### Core Architecture Components

*   **Frontend:** Next.js 14+ with App Router
*   **UI Framework:** React with Tailwind CSS
*   **Database:** Supabase (PostgreSQL)
*   **Authentication:** Next-Auth with multiple providers
*   **State Management:** React Context API and SWR for data fetching
*   **API Integration:** REST API routes for external services
*   **AI Processing:** OpenAI and Grok for analysis
*   **Deployment:** Vercel with serverless functions

### System Data Flow

The platform follows this primary data flow pattern:

1.   YouTube data is collected through the YouTube Data API
2.   Comments are analyzed for buyer intent using AI models
3.   High-potential comments are identified and stored
4.   Marketplace intelligence is gathered from various platforms
5.   Product concepts are validated against market data
6.   Designs are generated for validated concepts
7.   Mockups are created for visualization
8.   Results are presented in an interactive dashboard

### YouTube Data Synchronization

The platform synchronizes YouTube channel, playlist, video, and comment data with quota-aware scheduling.

#### Implementation Details

*   **API Endpoints:**`/api/youtube/sync/channels`, `/api/youtube/sync/playlists`, `/api/youtube/sync/videos`, `/api/youtube/sync/comments`
*   **Sync Engine:** Background jobs that respect YouTube API quotas with incremental updates
*   **Quota Management:** Intelligent scheduling system that distributes requests throughout the day
*   **Priority System:** Prioritizes channels and content based on engagement metrics

#### Code Implementation

// app/api/youtube/sync/channels/route.js
import { NextResponse } from 'next/server';
import { supabase } from '@/lib/supabaseClient';
import { YOUTUBE_API_KEY } from '@/lib/apiKeys';
import { getChannelData } from '@/lib/youtubeApi';
import { recordQuotaUsage } from '@/lib/quotaManager';

export async function POST(request) {
  try {
    const { channelIds, fullSync = false } = await request.json();
    
    // Quota check
    const quotaAvailable = await checkQuotaAvailability('channels', channelIds.length);
    if (!quotaAvailable) {
      return NextResponse.json({ 
        error: 'Quota limit approaching. Sync scheduled for later.' 
      }, { status: 429 });
    }
    
    const results = [];
    for (const channelId of channelIds) {
      // Get channel data from YouTube API
      const channelData = await getChannelData(channelId, YOUTUBE_API_KEY);
      
      // Record quota usage
      await recordQuotaUsage('youtube', 1);
      
      // Prepare data for database
      const channelRecord = {
        id: channelData.id,
        title: channelData.snippet.title,
        description: channelData.snippet.description,
        thumbnail_url: channelData.snippet.thumbnails.high.url,
        subscriber_count: channelData.statistics.subscriberCount,
        video_count: channelData.statistics.videoCount,
        last_updated: new Date().toISOString(),
        metadata: {
          publishedAt: channelData.snippet.publishedAt,
          country: channelData.snippet.country,
          customUrl: channelData.snippet.customUrl,
          viewCount: channelData.statistics.viewCount
        }
      };
      
      // Insert or update in database
      const { data, error } = await supabase
        .from('channels')
        .upsert(channelRecord, { onConflict: 'id' });
        
      if (error) throw error;
      
      results.push({ id: channelId, success: true });
    }
    
    return NextResponse.json({ results });
  } catch (error) {
    console.error('Channel sync error:', error);
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}

### Buyer Intent Detection System

AI-powered system for analyzing YouTube comments to identify purchase intent signals for print-on-demand products.

#### Implementation Details

*   **AI Models:** OpenAI for natural language understanding and Grok for specialized analysis
*   **Intent Patterns:** Detection of phrases like "I need that on a shirt", "would buy that", etc.
*   **Engagement Analysis:** Considers comment likes, replies, and viewer interaction patterns
*   **Score Calculation:** 0-100 scale based on multiple factors including language, engagement, and context

import { Configuration, OpenAIApi } from 'openai';

// Analyze a batch of comments for buyer intent
export async function analyzeBuyerIntent(comments) {
  const results = [];
  
  for (const comment of comments) {
    // Skip if already analyzed
    if (comment.buyer_intent_score) {
      results.push({
        id: comment.id,
        score: comment.buyer_intent_score,
        cached: true
      });
      continue;
    }
    
    try {
      // Prepare context for AI
      const videoContext = await getVideoContext(comment.video_id);
      
      // OpenAI analysis
      const response = await openai.createChatCompletion({
        model: "gpt-4",
        messages: [
          {
            role: "system",
            content: `You are analyzing YouTube comments to detect buyer intent for print-on-demand products. 
                     Score from 0-100 based on how strongly the comment indicates desire to purchase 
                     a physical product like a t-shirt, mug, poster, etc. Consider phrases like 
                     "I need this on a shirt", "would buy", "take my money", etc.`
          },
          {
            role: "user",
            content: `Video context: ${videoContext.title}
                     Comment: ${comment.text}
                     Likes: ${comment.like_count}
                     Replies: ${comment.reply_count}
                     
                     Analyze this comment for buyer intent. Provide:
                     1. Buyer intent score (0-100)
                     2. Reasoning for score
                     3. Potential product suggestion if applicable
                     4. Key phrases indicating purchase intent`
          },
        ],
      });
      
      // Parse AI response
      const analysis = parseAIResponse(response.data.choices[0].message.content);
      
      // Additional analysis with Grok for specialized insights
      if (analysis.score > 50) {
        const grokAnalysis = await analyzeWithGrok(comment, videoContext);
        // Merge insights from both models
        analysis.market_insights = grokAnalysis.market_insights;
        analysis.trend_alignment = grokAnalysis.trend_alignment;
      }
      
      // Update database with analysis
      const { error } = await supabase
        .from('comments')
        .update({ 
          buyer_intent_score: analysis.score,
          analysis: analysis
        })
        .eq('id', comment.id);
      
      if (error) throw error;
      
      results.push({
        id: comment.id,
        score: analysis.score,
        analysis: analysis
      });
    } catch (error) {
      console.error(`Error analyzing comment ${comment.id}:`, error);
      results.push({
        id: comment.id,
        error: error.message
      });
    }
  }
  
  return results;
}

### POD-Optimized Mockup Generator

Advanced system for generating product mockups from designs with multiple AI model options and template customization.

#### Implementation Details

*   **Multiple Design Models:** Support for simultaneous generation using various AI models (Ideogram, Midjourney, DALL-E)
*   **Product Types:** T-shirts, hoodies, mugs, wall art, phone cases, and more
*   **Prompt Engineering:** Specialized prompt templates for POD-optimized designs
*   **Sellerpic.ai Integration:** Professional mockup generation for final designs

#### Code Implementation

// app/api/design/generate/route.js
import { NextResponse } from 'next/server';
import { supabase } from '@/lib/supabaseClient';
import { generateDesign } from '@/lib/designGeneration';
import { createMockup } from '@/lib/mockupGeneration';
import { IDEOGRAM_API_KEY, OPENAI_API_KEY } from '@/lib/apiKeys';

export async function POST(request) {
  try {
    const { 
      prompt, 
      brandId = null, 
      commentId = null, 
      productType = 't-shirt',
      models = ['ideogram', 'dalle'] // Allow multiple model selection
    } = await request.json();
    
    // Get brand details if provided
    let brandDetails = null;
    if (brandId) {
      const { data, error } = await supabase
        .from('brands')
        .select('*')
        .eq('id', brandId)
        .single();
      
      if (error) throw error;
      brandDetails = data;
    }
    
    // Get comment details if provided
    let commentDetails = null;
    if (commentId) {
      const { data, error } = await supabase
        .from('comments')
        .select('*, videos(title)')
        .eq('id', commentId)
        .single();
      
      if (error) throw error;
      commentDetails = data;
    }
    
    // Generate designs using multiple models in parallel
    const designResults = await Promise.all(models.map(async (model) => {
      const enhancedPrompt = enhancePrompt(prompt, brandDetails, commentDetails, productType, model);
      
      const designResult = await generateDesign(enhancedPrompt, model);
      
      // Record design in database
      const { data: designData, error } = await supabase
        .from('designs')
        .insert({
          brand_id: brandId,
          title: generateTitle(prompt, model),
          description: prompt,
          prompt: enhancedPrompt,
          image_url: designResult.imageUrl,
          model_used: model,
          comment_id: commentId,
          metadata: {
            source: commentId ? 'comment' : 'manual',
            model_version: designResult.modelVersion,
            settings: designResult.settings
          }
        })
        .select()
        .single();
      
      if (error) throw error;
      
      return {
        designId: designData.id,
        model: model,
        imageUrl: designResult.imageUrl
      };
    }));
    
    // Generate mockup for first design (can be extended to all)
    const mockupResult = await createMockup(designResults[0].imageUrl, productType);
    
    // Record mockup in database
    const { data: mockupData, error } = await supabase
      .from('mockups')
      .insert({
        design_id: designResults[0].designId,
        product_type: productType,
        mockup_url: mockupResult.mockupUrl,
        service_used: 'sellerpic',
        settings: mockupResult.settings
      })
      .select()
      .single();
    
    if (error) throw error;
    
    return NextResponse.json({ 
      designs: designResults,
      mockup: {
        id: mockupData.id,
        url: mockupResult.mockupUrl
      }
    });
  } catch (error) {
    console.error('Design generation error:', error);
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}

// Helper to enhance prompts with context
function enhancePrompt(basePrompt, brandDetails, commentDetails, productType, model) {
  let enhancedPrompt = basePrompt;
  
  // Add product-specific instructions
  if (productType === 't-shirt') {
    enhancedPrompt += '. Optimize for t-shirt printing with clear design, limited colors, good contrast.';
  } else if (productType === 'wall-art') {
    enhancedPrompt += '. Create high-resolution art suitable for wall printing with rich details and visual appeal.';
  }
  
  // Add brand context if available
  if (brandDetails) {
    enhancedPrompt += ` Style should match ${brandDetails.name}'s branding with ${brandDetails.tone_of_voice} tone.`;
  }
  
  // Add video context if from comment
  if (commentDetails) {
    enhancedPrompt += ` Inspired by video: ${commentDetails.videos.title}.`;
  }
  
  // Model-specific optimizations
  if (model === 'ideogram') {
    enhancedPrompt += ' Detailed illustration, vibrant colors, professional quality.';
  } else if (model === 'dalle') {
    enhancedPrompt += ' Clear composition, printable design, vector-like quality.';
  }
  
  return enhancedPrompt;
}

// Generate title based on prompt
function generateTitle(prompt, model) {
  // Simple title generation logic
  const words = prompt.split(' ').slice(0, 5).join(' ');
  return `${words}... (${model})`;
}

// Log error to database and console
export async function logError(error, context = {}) {
  const {
    category = ErrorCategory.PROCESSING,
    severity = ErrorSeverity.ERROR,
    component = 'unknown',
    user = null,
    action = null
  } = context;
  
  // Format error data
  const errorData = {
    message: error.message,
    stack: error.stack,
    category,
    severity,
    component,
    user_id: user,
    action,
    context: JSON.stringify(context),
    timestamp: new Date().toISOString()
  };
  
  // Log to console
  console.error(`[${severity.toUpperCase()}] [${category}] ${error.message}`, {
    component,
    stack: error.stack,
    context
  });
  
  // Store in database
  try {
    await supabase.from('error_logs').insert(errorData);
  } catch (dbError) {
    // Fallback if database logging fails
    console.error('Failed to log error to database:', dbError);
  }
  
  // Alert for critical errors
  if (severity === ErrorSeverity.CRITICAL) {
    try {
      await sendAlertNotification(errorData);
    } catch (alertError) {
      console.error('Failed to send alert notification:', alertError);
    }
  }
  
  return errorData;
}

// Retry a function with exponential backoff
export async function withRetry(fn, options = {}) {
  const {
    maxRetries = 3,
    initialDelay = 1000,
    maxDelay = 10000,
    factor = 2,
    retryCondition = () =&gt; true
  } = options;
  
  let attempt = 0;
  let delay = initialDelay;
  
  while (true) {
    try {
      return await fn();
    } catch (error) {
      attempt++;
      
      // Check if we should retry
      if (attempt &gt;= maxRetries || !retryCondition(error)) {
        throw error;
      }
      
      // Log retry attempt
      await logError(error, {
        category: ErrorCategory.PROCESSING,
        severity: ErrorSeverity.WARNING,
        component: fn.name || 'unknown',
        action: `retry-attempt-${attempt}`,
        context: { maxRetries, attempt }
      });
      
      // Wait before retrying
      await sleep(delay);
      
      // Increase delay for next attempt (exponential backoff)
      delay = Math.min(delay * factor, maxDelay);
    }
  }
}

// Graceful degradation for API features
export function withFallback(primaryFn, fallbackFn, options = {}) {
  const {
    fallbackCondition = (error) =&gt; true,
    logFailure = true
  } = options;
  
  return async (...args) =&gt; {

// Circuit breaker pattern for external services
export function createCircuitBreaker(fn, options = {}) {
  const {
    threshold = 5,
    timeout = 30000,
    resetTimeout = 60000
  } = options;
  
  let failures = 0;
  let lastFailureTime = null;
  let state = 'CLOSED';
  
  return async (...args) =&gt; {
    // Check circuit state
    if (state === 'OPEN') {
      // Check if reset timeout has elapsed
      if (Date.now() - lastFailureTime &gt; resetTimeout) {
        state = 'HALF_OPEN';
      } else {
        throw new Error('Circuit breaker is open');
      }
    }
    
    try {
      const result = await fn(...args);
      
      // Reset on success
      if (state === 'HALF_OPEN') {
        state = 'CLOSED';
        failures = 0;
      }
      
      return result;
    } catch (error) {
      lastFailureTime = Date.now();
      
      // Increment failure count
      if (state === 'CLOSED' || state === 'HALF_OPEN') {
        failures++;
        
        // Open circuit if threshold is reached
        if (failures &gt;= threshold) {
          state = 'OPEN';
          
          setTimeout(() =&gt; {
            state = 'HALF_OPEN';
          }, timeout);
          
          await logError(error, {
            category: ErrorCategory.EXTERNAL_SERVICE,
            severity: ErrorSeverity.WARNING,
            component: fn.name || 'unknown',
            action: 'circuit-open',
            context: { failures, threshold }
          });
        }
      }
      
      throw error;
    }
  };
}
